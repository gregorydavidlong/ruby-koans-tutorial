def test_calling_global_methods_with_wrong_number_of_arguments
  exception = assert_raise(ArgumentError) do
    my_global_method
  end
  assert_match(%rwrong number of arguments/, exception.message)

  exception = assert_raise(ArgumentError) do
    my_global_method(1,2,3)
  end
  assert_match(%rwrong number of arguments/, exception.message)
end
