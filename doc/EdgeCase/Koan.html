def command_line(args)
  args.each do |arg|
    case arg
    when %r^-n\/(.*)\/$/
      @test_pattern = Regexp.new($1)
    when %r^-n(.*)$/
      @test_pattern = Regexp.new(Regexp.quote($1))
    else
      if File.exist?(arg)
        load(arg)
      else
        fail "Unknown command line argument '#{arg}'"
      end
    end
  end
end

def meditate
  setup
  begin
    send(name)
  rescue StandardError, EdgeCase::Sensei::AssertionError => ex
    failed(ex)
  ensure
    begin
      teardown
    rescue StandardError, EdgeCase::Sensei::AssertionError => ex
      failed(ex) if passed?
    end
  end
  self
end
